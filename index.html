Q.7 Write a function that accepts parameter n and returns factorial of n

//using while loop
function factorial(x){
    let ans=1;
    while(x>0)
    {
        ans = ans * x;
        x--;
    }
    console.log(ans);
}
factorial(5);

//using for loop
function facto(x){
    let factorial=1;
    for(let i=x; i>0; i--)
    {
        factorial*=i;
    }
    console.log(factorial);
}
facto(6);



//Output: 120 720
